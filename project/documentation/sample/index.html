    bool[,] CreateSimulationGrid(Board board) 
    {
        var width = board.Width;
        var height = board.Height;
        var grid = new bool[width, Height];

        for (int y = 0; y < height; y++)
        {
            for (int x = 0; x < width; x++)
            {
                grid[x, y] = board.IsOccupied(x, y);
            }
        }

        return grid;
    }
